De forma que... luego de varios intentos...

de ambos compiladores... (TC 3.0 y DJGPP)...
la solución más sencilla.... IMHO.....

sería la siguiente...

#include <stdio.h>

void main()
{
 int a,b;

 printf( "\n\nIngrese número entero A: ");
 scanf( "%i", &a );
 printf( "\nIngrese número entero B: ");
 scanf( "%i", &b );

 a=a+b,a=a-(b=(((a-b)>=b)*b)+((b>(a-b))*(a-b)));

 if( a==b )
  printf("\n\nSon números iguales.");
 else
  printf("\n\nEl mayor es %i y el menor es %i.", a, b );
}
